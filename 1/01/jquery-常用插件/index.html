				"autoUpdateInput": false,

				// 	"format": 'YYYYMMDD',

			    },

元素绑定插件动作的取消

如果插件没有提供 destroy() 这样的方式来“卸载”，可以用下面的方法来取消绑定在元素上的动作.unbind().removeData()

$('#keywords_input').daterangepicker({
				autoUpdateInput: false,
				locale: {
					cancelLabel: 'Clear'
				}
			});

// if not provide the destroy()
$('#keywords_input').unbind().removeData();

ajax 下载文件

https://github.com/johnculviner/jquery.fileDownload

这个必须要有 url 才能请求

如果服务器端返回的是“文件”而不是“下载链接”

http://www.henryalgus.com/reading-binary-files-using-jquery-ajax/

var req = new XMLHttpRequest();
req.open("POST", "/servicepath/Method?ids=" + ids, true);
req.responseType = "blob";
req.onreadystatechange = function () {
  if (req.readyState === 4 && req.status === 200) {

    // test for IE

    if (typeof window.navigator.msSaveBlob === 'function') {
      window.navigator.msSaveBlob(req.response, "PdfName-" + new Date().getTime() + ".pdf");
    } else {
      var blob = req.response;
      var link = document.createElement('a');
      link.href = window.URL.createObjectURL(blob);
      link.download = "PdfName-" + new Date().getTime() + ".pdf";

      // append the link to the document body

      document.body.appendChild(link);

      link.click();
    }
  }
};
req.send();
