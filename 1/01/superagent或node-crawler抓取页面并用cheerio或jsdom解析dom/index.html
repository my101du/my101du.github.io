superagent或node-crawler抓取页面，并用Cheerio或JSDOM解析DOM

 评论 字数统计: 1766字 阅读时长: 4分

crawling/scraping

  • node-crawer 爬虫工具 （或 superagent 请求代理模块+express）
  • 结合 cheerio / jsdom 解析爬到的 html 内容(可以认为是 Node.js 版的 jquery)

node-crawer （推荐）

安装

npm install crawler

Basic Usage

开始

先引入 library require("crawler")； 然后根据传入参数初始化一个实例 var c = new Crawler({...})； 接着调用实例的方法 c.queue(...)。

var Crawler = require("crawler");

var c = new Crawler({
    maxConnections : 10,
    // This will be called for each crawled page
    callback : function (error, res, done) {
        if(error){
            console.log(error);
        }else{
            var $ = res.$;
            // $ is Cheerio by default
            //a lean implementation of core jQuery designed specifically for the server
            console.log($("title").text());
        }
        done();
    }
});

// Queue just one URL, with default callback
c.queue('http://www.amazon.com');

// Queue a list of URLs
c.queue(['http://www.google.com/','http://www.yahoo.com']);

// Queue URLs with custom callbacks & parameters
c.queue([{
    uri: 'http://parishackers.org/',
    jQuery: false,

    // The global callback won't be called
    callback: function (error, res, done) {
        if(error){
            console.log(error);
        }else{
            console.log('Grabbed', res.body.length, 'bytes');
        }
        done();
    }
}]);

// Queue some HTML code directly without grabbing (mostly for tests)
c.queue([{
    html: '<p>This is a <strong>test</strong></p>'
}]);

限制多个请求之间的频率（例如有的服务器限制了不能同时多个请求）

rateLimit指定后，在两个任务之间, 最小时间间隔为 1000ms = 1s

var c = new Crawler({
    rateLimit: 1000, // `maxConnections` will be forced to 1
    callback: function(err, res, done){
        console.log(res.$("title").text());
        done();
    }
});

c.queue(tasks);//between two tasks, minimum time gap is 1000 (ms)

请求 URL 带自定义参数(例如代码里要动态调整参数的值)

c.queue({
    uri:"http://www.google.com",
    parameter1:"value1",
    parameter2:"value2",
    parameter3:"value3"
});

//打印出来
console.log(res.options.parameter1);

返回原始内容而不是字符串（例如下载图片、pdf文件等）

通常是返回字符串的，但如果需要实现“下载文件”，则参数要改成 encoding:null ,并调用 fs模块处理文件下载

var fs = require('fs');

var c = new Crawler({
	// 直接下载
    encoding:null,

    jQuery:false,// set false to suppress warning message.
    callback:function(err, res, done){
        if(err){
            console.error(err.stack);
        }else{
        	// 把文件存储下来
            fs.createWriteStream(res.options.filename).write(res.body);
        }
        
        done();
    }
});

c.queue({
    uri:"https://nodejs.org/static/images/logos/nodejs-1920x1200.png",
    filename:"nodejs-1920x1200.png"
});

在发起请求前运行一些动作

在同步/异步请求发起之前进行一些动作，可以设置为全局的，也可以在单个请求里覆盖它

var c = new Crawler({
    preRequest: function(options, done) {
        // 'options' here is not the 'options' you pass to 'c.queue', instead, it's the options that is going to be passed to 'request' module 
        console.log(options);
	// when done is called, the request will start
	done();
    },
    callback: function(err, res, done) {
        if(err) {
	    console.log(err)
	} else {
	    console.log(res.statusCode)
	}
    }
});

c.queue({
    uri: 'http://www.google.com',
    // this will override the 'preRequest' defined in crawler
    preRequest: function(options, done) {
        setTimeout(function() {
	    console.log(options);
	    done();
	}, 1000)
    }
});

高级

直接 direct() 发起请求，而不是用queue()

如果不想经过 node-crawer 的调度器（Scheduler）,它的参数 options 和 queue() 是类似的。只不过 callback 的参数有点差异（error, response）

crawler.direct({
    uri: 'http://www.google.com',
    skipEventRequest: false, // default to true, direct requests won't trigger Event:'request'
    callback: function(error, response) {
        if(error) {
            console.log(error)
        } else {
            console.log(response.statusCode);
        }
    }
});

控制请求“阀门”（频率等）

前面提到过，在初始化 Crawler 对象时，用 rateLimit 限制两个请求之间的时间间隔；用 maxConnections 限制同时运行几个任务。 当 rateLimit 设置为大于0，maxConnnction 自动设置成 1

这里limiter参数，可以对使用 proxy 的请求进行控制

var crawler = require('crawler');

var c = new Crawler({
    rateLimit: 2000,
    maxConnections: 1,
    callback: function(error, res, done) {
        if(error) {
            console.log(error)
        } else {
            var $ = res.$;
            console.log($('title').text())
        }
        done();
    }
})

// if you want to crawl some website with 2000ms gap between requests
c.queue('http://www.somewebsite.com/page/1')
c.queue('http://www.somewebsite.com/page/2')
c.queue('http://www.somewebsite.com/page/3')

// if you want to crawl some website using proxy with 2000ms gap between requests for each proxy
c.queue({
    uri:'http://www.somewebsite.com/page/1',
    limiter:'proxy_1',
    proxy:'proxy_1'
})
c.queue({
    uri:'http://www.somewebsite.com/page/2',
    limiter:'proxy_2',
    proxy:'proxy_2'
})
c.queue({
    uri:'http://www.somewebsite.com/page/3',
    limiter:'proxy_3',
    proxy:'proxy_3'
})
c.queue({
    uri:'http://www.somewebsite.com/page/4',
    limiter:'proxy_1',
    proxy:'proxy_1'
});

除了在初始化的时候指定值，还能用下面的方法动态修改

var c = new Crawler({});
c.setLimiterProperty('limiterName', 'propertyName', value)；

Crawler 类

schedule 事件

当任务被添加到调度器时被触发

  • options Options
crawler.on('schedule',function(options){
    options.proxy = "http://proxy:port";
});

limiterChange 事件

  • options Options
  • limiter String

request 事件

准备发送请求时

  • options Options
crawler.on('request',function(options){
    options.qs.timestamp = new Date().getTime();
});

drain 事件

队列为空时

crawler.on('drain',function(){
    // For example, release a connection to database.
    db.end();// close connection to MySQL
});

crawler.queue(uri|options)

将任务放入队列等待执行

crawler.queueSize

获取队列大小

Options 参考

参考 https://github.com/request/request#requestoptions-callback

基本

options.uri: String The url you want to crawl. options.timeout : Number In milliseconds (Default 15000).

callback(error, res, done)

error: (Error)

res

这个很重要，它返回了 $这个 jQuery 选择器，后面解析 DOM 用到

返回了一个标准的 http.IncomingMessage 包括

  • res.statusCode （Number 状态码）
  • res.body (Buffer String) HTTP response content 如 html 页面，纯文本，xml 等（和服务器返回数据有关）
  • res.headers (Object)HTTP response headers
  • res.request
    • res.request.uri
    • res.request.method
    • res.request.headers
  • res.options (Options)
  • $

done (Function)

在完成工作后，必须在 callback() 里调用它

Schedule options

  • options.maxConnections: Number Size of the worker pool (Default 10).
  • options.rateLimit: Number Number of milliseconds to delay between each requests (Default 0).
  • options.priorityRange: Number Range of acceptable priorities starting from 0 (Default 10).
  • options.priority: Number Priority of this request (Default 5). Low values have higher priority.

Retry options

  • options.retries: Number Number of retries if the request fails (Default 3),
  • options.retryTimeout: Number Number of milliseconds to wait before retrying (Default 10000),

Server-side DOM options(jQuery)

  • options.jQuery (Boolean)

默认使用 cheerio 库来注入document 如果设置为 FALSE 则关闭了注入 jQuery 选择器

Charset encoding

  • options.forceUTF8
  • options.incomingEncoding

Cache

  • options.skipDuplicates (Boolean)

如果设置为 true, 则自动跳过已被爬过的 URI, 甚至都不会调用 callback(). （默认是 false）

不过不推荐设置它，而应该使用 https://github.com/mike442144/seenreq 来判断处理请求是否已经完成过。

Http headers

  • options.rotateUA (Boolean)
  • options.userAgent (Array)
  • options.referer (String) If truthy sets the HTTP referer header
  • options.headers (Object) Raw key-value of http headers

和 Cheerio or JSDOM 结合

Crawler 默认用 Cheerio（可能比较轻量），如果要用更强大的 JSDOM 需要提前引入 require('jsdom')

var jsdom = require('jsdom');
var Crawler = require('crawler');

var c = new Crawler({
    jQuery: jsdom
});

有下面几种方法来配置使用 cheerio 作为 jQuery 选择器使用的库

var c = new Crawler({
	jQuery: true //(default)
	//OR
	jQuery: 'cheerio'
	//OR
	jQuery: {
	    name: 'cheerio',
	    options: {
	        normalizeWhitespace: true,
	        xmlMode: true
	    }
	}
});

选项都来自 htmlparser2

这些是默认选项，你可以覆盖它们

{
    normalizeWhitespace: false,
    xmlMode: false,
    decodeEntities: true
}

拿到返回结果后，就可以用 cheerio 来解析了(完整的见后面)，例如

// sres.text 里面存储着网页的 html 内容，将它传给 cheerio.load 之后
// 就可以得到一个实现了 jquery 接口的变量，我们习惯性地将它命名为 `$`
// 剩下就都是 jquery 的内容了
var $ = cheerio.load(sres.text);
var items = [];
$('#topic_list .topic_title').each(function (idx, element) {
var $element = $(element);
items.push({
  title: $element.attr('title'),
  href: $element.attr('href')
});
});

res.send(items);
