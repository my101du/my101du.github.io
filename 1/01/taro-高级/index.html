dva.js

import {create} from 'dva-core';
import {createLogger} from 'redux-logger';
import createLoading from 'dva-loading';
let app;
let store;
let dispatch;

function createApp(opt) {
  opt.onAction = [createLogger()];
  app = create(opt);
  app.use(createLoading({}));

  if (!global.registered) opt.models.forEach(model => app.model(model));
  global.registered = true;
  app.start();

  store = app._store;
  app.getStore = () => store;

  dispatch = store.dispatch;

  app.dispatch = dispatch;
  return app;
}

export default {
  createApp,
  getDispatch() {
    return app.dispatch;
  }
}

util/request.js

import Taro from '@tarojs/taro'

export default function request(opt) {
  return Taro.request(opt).then((res) => {
    let {statusCode, data} = res;
    if (statusCode >= 200 && statusCode < 300) {
      return data;
    } else {
      throw new Error(`网络请求错误，状态码${statusCode}`);
    }
  })
}

Error: Node Sass does not yet support your current environment

在一次 taro update project 后， 再运行 npm run dev:h5，爆出这个错误。

完整的信息类似

can not build .temp/app.scss

Error: Node Sass does not yet support your current environment： MacOS xxxxxx (67) please check the node-sass 4.9.x （一个连接地址）

查了一大堆资料， node重装，手动 node rebuild node-sass等等都不行。

后来根据提示的那个关键 4.9.x，编辑 node_modules/sass-loader/package.json, 把里面一段

"node-sass": "^4.5.0",

改成

"node-sass": "^4.9.0",

在这个 sass-loader 目录里面 npm install 等待很久。

回到项目目录，npm run dev:h5 成功了。

估计是开发了很久没有去升级 taro update project, 突然升级下，而我的 macos 又升级了大版本从 10.13->10.14. node-sass 4.5.x 太旧，无法兼容新的 macOS, 而taro 还没有更新里面那个版本号。。。。

但是查看 node_moduels/@tarojs/plugin-sass/package.json 里

只能说不知什么原因 sass-loader 的版本已经到了最新的 7.1.0，但是它 package.json 里使用的 node-sass 4.5.x 没升级。

https://github.com/webpack-contrib/sass-loader/blob/master/package.json

果然。。。

Error: 404 response downloading https://r.cnpmjs.org/node/v11.1.0/node-v11.1.0-headers

提示 node headers 一直 404 的问题， (不管是改成 node 8 还是 node 10 都是 404)

查看 ~/.npmrc，咦，奇怪，我明明设置了 npm 去淘宝的源 registry.npm.taobao.org，怎么还是 r.cnpmjs.org ? （虽然淘宝的源也是从后者自动去拉取同步的。。。。）

# NPM Taobao registry
registry=https://registry.npm.taobao.org

打开项目目录下的 .npmrc 文件, 赫然发现在这里覆盖了系统的参数。。。（r.cnpmjs.org）

把 .npmrc 里的内容全部改成 npm.taobao.org

再次运行 npm install，咦，还是报错，那个 node-sass 的问题

最后在此处找到答案 https://github.com/NervJS/taro/issues/1034

首先注释项目目录下 .npmrc 里的内容， 修改 ~/.npmrc 添加两行。

特别注意第二个地址，前面有cdn.

# Taobao node binary mirrors
fse_binary_host_mirror=https://npm.taobao.org/mirrors/fsevents
sass_binary_site=http://cdn.npm.taobao.org/dist/node-sass

重新npm install,不再报错了，这次用的是一个新的地址 cdn.npm.taobao.org 来下载 node-sass
