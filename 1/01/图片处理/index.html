public function setMemoryLimit($filename)
{
    //this might take time so we limit the maximum execution time to 50 seconds
    set_time_limit(50);

    //reset to the default value
    $defaultMemoryLimit = "256";
    ini_set('memory_limit', $defaultMemoryLimit . 'M');

    //getting the image width and height
    $width = 0;
    $height = 0;
    list($width, $height) = getimagesize($filename);
    
    $size = $defaultMemoryLimit + floor(($width * $height * 4 * 1.5 + 1048576) / 1048576);

    //We don't want to allocate an extremely large amount of memory, define a limit
    $maxAllowedMemory = "1024";
    
    if ($size > $maxAllowedMemory) {
        $size = $maxAllowedMemory;
    }

    //updating the default value
    ini_set('memory_limit', $size.'M');
    echo 'allocate new memory size:' . $size.'M<br />';
}
